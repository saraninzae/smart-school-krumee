const video = document.getElementById('video');
const statusText = document.getElementById('status');

// 🔴 วาง URL ของ Google Apps Script (Web App) ของคุณที่นี่
const GAS_WEB_APP_URL = "https://script.google.com/macros/s/AKfycbx7sgOblfDdXG_rdVacAFV6_AObBY2iTz5JPBl9PcBf5jetJflWEnifbJS8gCPTwveB6A/exec";

// ตำแหน่งที่เก็บไฟล์ Model AI ของ face-api
const MODEL_URL = 'https://justadudewhohacks.github.io/face-api.js/models';

// 1. โหลดโมเดล AI ให้พร้อมทำงาน
Promise.all([
    faceapi.nets.ssdMobilenetv1.loadFromUri(MODEL_URL),
    faceapi.nets.faceLandmark68Net.loadFromUri(MODEL_URL),
    faceapi.nets.faceRecognitionNet.loadFromUri(MODEL_URL)
]).then(startVideo);

// 2. ฟังก์ชันขอสิทธิ์เปิดกล้อง
function startVideo() {
    statusText.innerText = "กำลังเชื่อมต่อกล้อง...";
    navigator.mediaDevices.getUserMedia({ video: {} })
        .then(stream => {
            video.srcObject = stream;
            statusText.innerText = "กล้องพร้อมใช้งานแล้ว! กรุณามองที่กล้อง";
        })
        .catch(err => {
            statusText.innerText = "ไม่สามารถเปิดกล้องได้ กรุณาตรวจสอบสิทธิ์การเข้าถึงกล้องในเบราว์เซอร์";
            statusText.style.color = "red";
            console.error(err);
        });
}

// 3. เริ่มสแกนเมื่อวิดีโอเล่น
video.addEventListener('play', async () => {
    // สร้าง Canvas เพื่อวาดกรอบสี่เหลี่ยมทับใบหน้า
    const canvas = faceapi.createCanvasFromMedia(video);
    document.getElementById('video-container').append(canvas);
    const displaySize = { width: video.width, height: video.height };
    faceapi.matchDimensions(canvas, displaySize);

    let isScanning = false; // ตัวแปรป้องกันการส่งข้อมูลซ้ำรัวๆ

    setInterval(async () => {
        if (isScanning) return; 

        // ตรวจจับใบหน้า
        const detections = await faceapi.detectAllFaces(video).withFaceLandmarks().withFaceDescriptors();
        const resizedDetections = faceapi.resizeResults(detections, displaySize);
        
        // ลบกรอบเก่า และวาดกรอบใหม่
        canvas.getContext('2d').clearRect(0, 0, canvas.width, canvas.height);
        faceapi.draw.drawDetections(canvas, resizedDetections);

        // ถ้าเจอใบหน้าอย่างน้อย 1 หน้า
        if (detections.length > 0) {
            isScanning = true;
            statusText.innerText = "ตรวจพบใบหน้า กำลังบันทึกข้อมูล...";
            statusText.style.color = "green";

            // *** สำหรับช่วงทดสอบระบบ เราจะจำลองการส่งชื่อไปก่อน ***
            // ในระบบจริง เราจะเอา Descriptor ไปเทียบหาว่าเป็นใบหน้าของใคร
            let personName = "ครูมี (Saranin)"; 

            recordAttendance(personName);
            
            // หน่วงเวลา 5 วินาที ก่อนเปิดให้สแกนคนต่อไปได้
            setTimeout(() => {
                isScanning = false;
                statusText.innerText = "กล้องพร้อมใช้งานแล้ว! กรุณามองที่กล้อง";
                statusText.style.color = "#e67e22";
                canvas.getContext('2d').clearRect(0, 0, canvas.width, canvas.height);
            }, 5000);
        }
    }, 1000); // เช็คใบหน้าทุกๆ 1 วินาที
});

// 4. ฟังก์ชันส่งข้อมูลไปยัง Google Sheets และ Telegram
function recordAttendance(personName) {
    const payload = JSON.stringify({
        name: personName,
        action: "เข้างาน"
    });

    fetch(GAS_WEB_APP_URL, {
        method: 'POST',
        headers: {
            'Content-Type': 'text/plain;charset=utf-8', 
        },
        body: payload
    })
    .then(response => response.json())
    .then(data => {
        console.log("บันทึกสำเร็จ:", data);
        // สามารถเพิ่มเสียงแจ้งเตือน หรือ popup ตรงนี้ได้
    })
    .catch((error) => {
        console.error("Error:", error);
        statusText.innerText = "เกิดข้อผิดพลาดในการส่งข้อมูลไปที่ระบบฐานข้อมูล";
        statusText.style.color = "red";
    });
}
